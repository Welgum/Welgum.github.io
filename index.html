min_value = np.amin(pneum)
max_value = np.amax(pneum)
vrange = max_value - min_value
h = (max_value - min_value) / (1 + 3.22 * np.log10(len(pneum)))
k = int(np.floor((max_value - min_value) / h))
print('Шаг: ', h)
print('Кол-во интервалов: ', k)
pneum_intervals = [i * 0 for i in range(k)]
pneum_by_groups = [[] for i in range(k)]
for val in pneum:
    #print(str(int(val / h)) + " : " + str(val))
    if int(val / h) < k:
        pneum_intervals[int(val / h)] += 1
        pneum_by_groups[int(val / h)].append(val)
